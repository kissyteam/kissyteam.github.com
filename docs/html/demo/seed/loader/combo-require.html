配置依赖 combo 使用 gallery¶

KISSY.config('modules',{
    "gallery/browsercache/1.0/":{
        requires:["gallery/local-storage/1.0/"]
    }
});
KISSY.use("gallery/browsercache/1.0/", function () {
    alert("browsercache is loaded!");
});
